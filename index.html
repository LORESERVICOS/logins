const $ = (id) => document.getElementById(id); // “Apagar o login” ao abrir (não mantém sessão) sessionStorage.removeItem("auth_ok"); $("btn").onclick = async () => { $("msg").textContent = "Validando..."; const usuario = $("usuario").value.trim(); const senha = $("senha").value.trim(); if (!usuario || !senha) { $("msg").textContent = "Preencha usuário e senha."; return; } try { const resp = await fetch(WEBAPP_URL, { method: "POST", headers: {"Content-Type":"application/json"}, body: JSON.stringify({ tipo: "login", usuario, senha }) }); const data = await resp.json(); if (data.status !== "ok") { $("msg").textContent = data.message || "Login inválido"; // “apaga” os campos pra ter que digitar de novo $("senha").value = ""; return; } // Login OK: não salvar login permanente. // Se quiser uma trava mínima pra página, salva só na sessão atual: sessionStorage.setItem("auth_ok", "1"); // limpa campos (o “apagado” que você pediu) $("usuario").value = ""; $("senha").value = ""; // Redireciona window.location.href = data.redirect || "pagina.html"; } catch (e) { $("msg").textContent = "Erro de conexão/URL do WebApp."; console.error(e); } };
